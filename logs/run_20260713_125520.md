# Load the frames + camera, and PROVE the eval harness with two model-free self-checks.
man = json.loads(Path('data/manifest.json').read_text())
cam = man['camera']
f, cx, cy = cam['f'], cam['cx'], cam['cy']
cam_pos = np.array(cam['cam_pos'], float)
R_cw    = np.array(cam['R_cw'], float)
FRAMES  = man['frames']
print(len(FRAMES), 'frames | %dx%d | f=%.1f' % (cam['width'], cam['height'], f))

def project(P):
    # World point -> (u, v, forward-axis depth m). Inverse of point_at_depth.
    xc, yc, zc = R_cw.T @ (np.asarray(P, float) - cam_pos)
    return cx + f * xc / -zc, cy - f * yc / -zc, -zc

# Check A (the eval math): project gt_xyz then deproject at that depth -> recover gt_xyz.
# Pure inverse, full precision, no model, no manifest rounding -> machine epsilon.
inv = []
for fr in FRAMES:
    P = np.array(fr['gt_xyz'], float); u, v, d = project(P)
    inv.append(np.linalg.norm(CM.point_at_depth(u, v, f, cx, cy, cam_pos, R_cw, d) - P))
print('self-check A  project<->deproject  max err = %.2e m  (want ~1e-16)' % max(inv))

# Check B (the stored data): reprojected centre must match the manifest's stored gt_uv.
px = [np.hypot(*(np.subtract(project(fr['gt_xyz'])[:2], fr['gt_uv']))) for fr in FRAMES]
print('self-check B  stored gt_uv vs reproj  max err = %.4f px' % max(px))

# SCALE ANCHOR — the robot BASE is the origin (0,0,0), a fixed KNOWN point (base==world here).
# Its pixel and true forward-axis depth are analytic. Reading a depth model's value at that pixel
# gives the global scale with NO cube/plane GT — honest robot self-knowledge only (like a live FK
# gripper anchor, but the base never moves and is always in view).
BASE_UV = tuple(project([0.0, 0.0, 0.0])[:2])
BASE_DEPTH = float(project([0.0, 0.0, 0.0])[2])
print('scale anchor: base pixel (%.1f, %.1f)  true depth %.4f m' % (BASE_UV[0], BASE_UV[1], BASE_DEPTH))
```

# One function per model: RGB image (H,W,3 uint8) -> (H,W) metric depth in metres,
# along the camera forward axis (the convention camera_math.point_at_depth consumes).
import torch
_DEV = 'cuda' if torch.cuda.is_available() else 'cpu'
print('device:', _DEV)
_M = {}   # load each model once

def depth_pro_map(img_rgb):
    import depth_pro, os
    if 'dp' not in _M:
        ckpt = 'checkpoints/depth_pro.pt'          # create_model_and_transforms reads ./checkpoints/depth_pro.pt
        if not (os.path.exists(ckpt) and os.path.getsize(ckpt) > 10_000_000):
            os.makedirs('checkpoints', exist_ok=True)
            print('Depth Pro checkpoint missing -> downloading (~1.9 GB, one-time)...')
            rc = os.system('wget -q https://ml-site.cdn-apple.com/models/depth-pro/depth_pro.pt -O ' + ckpt)
            print('  wget rc=%d  size=%.0f MB' % (rc, os.path.getsize(ckpt) / 1e6 if os.path.exists(ckpt) else 0))
        m, tf = depth_pro.create_model_and_transforms(device=torch.device(_DEV))
        _M['dp'] = (m.eval(), tf)
    m, tf = _M['dp']
    with torch.no_grad():
        # f_px MUST be a device tensor: Depth Pro's infer() runs f_px.squeeze() internally,
        # which throws on a Python float. Passing the known focal skips its focal estimation.
        pred = m.infer(tf(img_rgb), f_px=torch.tensor(float(f), device=_DEV))
    return pred['depth'].squeeze().float().cpu().numpy()

def unidepth_map(img_rgb):
    from unidepth.models import UniDepthV2
    if 'ud' not in _M:
        m = UniDepthV2.from_pretrained('lpiccinelli/unidepth-v2-vitl14')
        m.resolution_level = 9          # [0,10] accuracy knob; unset -> coarse default bounds (worse depth)
        _M['ud'] = m.to(_DEV).eval()
    rgb = torch.from_numpy(img_rgb).permute(2, 0, 1)                 # C,H,W uint8
    K = torch.tensor([[f, 0, cx], [0, f, cy], [0, 0, 1]], dtype=torch.float32)
    with torch.no_grad():
        pred = _M['ud'].infer(rgb, K)
    return pred['depth'].squeeze().float().cpu().numpy()

# Score a model. Per frame: read its depth map ONCE, take depth at the KNOWN cube pixel and at
# the KNOWN base pixel. Report loc error three ways:
#   raw            - no scale (shows the model's native metric error)
#   oracle_scaled  - one global scale = median(gt_depth/pred) over ALL frames (CHEATS; upper bound)
#   base_anchored  - HONEST: per-frame scale = true_base_depth / model_depth_at_base_pixel
# The base_anchored column is the real number; oracle is how close a perfect scale could get.
def _patch(dmap, u, v, r=2):
    v0, v1 = max(0, int(round(v)) - r), int(round(v)) + r + 1
    u0, u1 = max(0, int(round(u)) - r), int(round(u)) + r + 1
    return float(np.median(dmap[v0:v1, u0:u1]))

def eval_model(depth_map, name, show=6):
    recs = []
    for fr in FRAMES:
        dmap = depth_map(np.array(Image.open('data/' + fr['png']).convert('RGB')))
        u, v = fr['gt_uv']
        d = float(dmap[int(round(v)), int(round(u))])
        db = _patch(dmap, BASE_UV[0], BASE_UV[1])     # model depth at the fixed base pixel
        recs.append((u, v, d, db, np.array(fr['gt_xyz'], float), fr['gt_depth_m'], fr['png']))
    def loc(scale_of):
        return np.array([np.linalg.norm(CM.point_at_depth(u, v, f, cx, cy, cam_pos, R_cw, d * scale_of(d, db)) - g)
                         for (u, v, d, db, g, gd, pp) in recs]) * 1000
    oracle = float(np.median([gd / d for (u, v, d, db, g, gd, pp) in recs if d > 0]))
    raw  = loc(lambda d, db: 1.0)
    orc  = loc(lambda d, db: oracle)
    anch = loc(lambda d, db: (BASE_DEPTH / db) if db > 0 else 1.0)
    print('  %s: per-frame depths + base-anchor scale (first %d)' % (name, show))
    for u, v, d, db, g, gd, pp in recs[:show]:
        print('    %-16s gt %.3f  cube_pred %.3f  base_pred %.3f  base_scale %.2f'
              % (pp, gd, d, db, (BASE_DEPTH / db if db else 0)))
    r = {'n': len(recs),
         'raw_median_mm': round(float(np.median(raw)), 1),
         'oracle_scaled_median_mm': round(float(np.median(orc)), 1),
         'oracle_scaled_p95_mm': round(float(np.percentile(orc, 95)), 1),
         'BASE_ANCHORED_median_mm': round(float(np.median(anch)), 1),
         'BASE_ANCHORED_p95_mm': round(float(np.percentile(anch, 95)), 1),
         'median_base_scale': round(float(np.median([BASE_DEPTH / db for (u, v, d, db, g, gd, pp) in recs if db > 0])), 3)}
    print('  ->', name, r)
    return r

results = {}
def _run(name, fn):
    try:
        results[name] = eval_model(fn, name)
    except Exception as e:      # keep the other model's number if one backend fails
        results[name] = {'error': '%s: %s' % (type(e).__name__, str(e)[:120])}
        print('%-12s FAILED: %s' % (name, results[name]['error']))

_run('depth_pro', depth_pro_map)        # ACTIVE this session
# _run('unidepth_v2', unidepth_map)     # uncomment instead if you installed UniDepth
```

# natively metric (ratio ~1.0) on the hero frame. Cheap: 3 inferences on one image.
import torch
if 'dp' in _M:
    m, tf = _M['dp']
    fr0 = min(FRAMES, key=lambda fr: abs(fr['gt_xyz'][1]))
    img0 = np.array(Image.open('data/' + fr0['png']).convert('RGB'))
    u0, v0 = fr0['gt_uv']; x0 = tf(img0); gd0 = fr0['gt_depth_m']
    trials = [('f=%.1f native-res' % f, float(f)),
              ('f=None (estimated)', None),
              ('f=%.1f (f*1536/640)' % (f * 1536 / 640), float(f) * 1536 / 640)]
    for tag, fpx in trials:
        kw = {} if fpx is None else {'f_px': torch.tensor(float(fpx), device=_DEV)}
        with torch.no_grad():
            dm = m.infer(x0, **kw)['depth'].squeeze().float().cpu().numpy()
        d0 = float(dm[int(round(v0)), int(round(u0))])
        print('  %-26s pred %.3f m  gt %.3f m  ratio %.2f' % (tag, d0, gd0, (gd0 / d0 if d0 else 0)))
else:
    print('focal probe skipped (Depth Pro not loaded this session)')
